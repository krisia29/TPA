Export to Google Sheets

Push live student progress into a Google Sheet with one click using Apps Script.

Step 1 — Create your spreadsheet

Open Google Sheets, create a new file, then go to Extensions → Apps Script.

Step 2 — Paste this script

Delete existing code and paste:

function doPost(e) { var ss = SpreadsheetApp.getActiveSpreadsheet(); var sheet = ss.getActiveSheet(); var data = JSON.parse(e.postData.contents); var tools = data.tools; sheet.clearContents(); var header = ['Name','Email','Joined','Total %','Done', 'Total Activities','Level','Verified Uploads'].concat( tools.map(function(t){return t+' %';})).concat( tools.map(function(t){return t+' done';})).concat( tools.map(function(t){return t+' verified';})); sheet.appendRow(header); data.students.forEach(function(s){ var row=[s.name,s.email,s.joined,s.pct,s.done, s.total,s.level,s.verified]; tools.forEach(function(t){row.push(s.tools[t]||0);}); tools.forEach(function(t){row.push(s.toolDone[t]||0);}); tools.forEach(function(t){row.push(s.toolVerified[t]||0);}); sheet.appendRow(row); }); return ContentService.createTextOutput( JSON.stringify({status:'ok'})) .setMimeType(ContentService.MimeType.JSON); }
Step 3 — Deploy as Web App

Click Deploy → New deployment. Type = Web App. Access = Anyone. Copy the URL.
